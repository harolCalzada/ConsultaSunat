# *****************************************************************
# Este es el archivo de configuración que indica al programa donde
# encontrar la información de las convocatorias
# 
# Cada línea que contiene datos debe contener un indicador del dato
# que se va a definir y el dato encerrado en barras verticales (|)
# Dependiendo del centro de empleo, algunos datos son opcionales
# Por ejemplo, si la página no lo soporta.
# Para definir un dato opcional, se agrega al inicio (entre las
# barras verticales) la palabra 'optional'. 
# Si un dato es opcional, no es necesario definirlo:
# Nombre del Valor:         |optional|
# 
# La estructura que se espera de las convocatorias (la que el 
# programa sigue para obtener los datos) es esta:
# Por cada área de desempeño, se busca por cada periodo de 
# publicación, todas las convocatorias.
# 
# Los datos de tipo HTML (estructura donde encontrar ciertos datos)
# siguen algunas reglas:
# El símbolo '->' define un nivel más profundo de jerarquía
# Cada etiqueta (nivel) tiene dos partes, la ocurrencia de la 
# etiqueta que se busca 
# (1 (primera), 2 (segunda), * (todas las ocurrencias), etc) }
# y el detalle de la etiqueta.
# Este detalle consiste de tres partes: 
# el nombre de la etiqueta (ul, div, * (todas las etiquetas), etc)
# las características de la etiqueta en forma de diccionario 
# ie. { class='job_box' }
# El atributo que se desea obtener (vacío si se desea obtener
# el contenido de la etiqueta)
# ie. href
# Ejemplo completo:
# 1 ul\{"data-key":"areas"}\->* label\{}\->1 input\{}\value
# Se obtiene la primera lista no ordenada cuyo atributo 'data-key' 
# tenga el valor 'areas'. 
# En su contenido se buscan todos los label
# (diccionario vacío => sin restricción). 
# Dentro de cada label se busca el primer input y se obtiene el 
# valor de su atributo 'value'
# 
# Además, se define un tipo de dato lista
# *****************************************************************
#
# Primero se indica el inicio de la sección de datos globales (que conciernen a todos las convocatorias o asisten la búsqueda)
#
Global:
#
# Luego se indica el nombre del centro de empleo
# Este se utiliza para el reporte y para el keyspace de la base de 
# datos donde se guardarán las convocatorias recogidas
#
Job Center:                 |Example|
#
# El nombre del archivo de funciones específicas del centro de 
# empleo, para formar el enlace absoluto de oferta, etc 
# A este nombre se le añade la extensión .py y se busca en la
# carpeta Functions
#
Functions Filename:         |example|
#  
# El url base a partir del cual se
# formarán los enlaces de las áreas, periodos y ofertas
#
URL Base:                   |http://example-jobs.com/lima-peru
# 
# El periodo de publicación cuyas convocatorias se extraerán
# 
Period:                     |hoy|
# 
# El url en el que se encuentran las áreas de trabajo 
# 
Areas URL:                  |http://example-jobs.com/lima-peru/areas|
# 
# La estructura de HTML donde se encuentran los enlaces a las áreas
# 
Areas Source:               |1 ul\{"data-key":"areas"}\->* label\{}\->1 input\{}\value|
# 
# Estructura donde se halla el número de ofertas encontradas
# 
Number of Offers Source:    |optional|
# 
# Número de links por página, si la página soporta paginación
# 
Links per Page:             |optional|
# 
# Número de fuentes de las cuales se pueden obtener enlaces a
# las convocatorias
# 
Number of Sources:          |1|
# 
# Estructuras donde encontrar las convocatorias (deben coincidir con el Number of Sources)
# 
Offer Sources:              |* a\{"class":"jobs_ads"}\href->1 div\{"class":"jobs_ads_options"}\->1 span\{}\|
#
# Feature a partir del cual obtener la fecha de publicación si no se encuentra definida en alguna de las estructuras 
# previamente especificadas. Dato tipo texto
#
Date Feature:               |optional|
# 
# Número de features en primer nivel (antes de entrar a la página
# de la convocatoria)
# 
First Level Features:       |1|
#
# Es necesario especificar cada feature de primer nivel por su cuenta (Un par Feature Name, Feature Values)
# Es menos eficiente, pero permite mejor control en este caso
# 
Feature Name:              |["ubicación"]|
#
# En esta fuente, la iteración (*) significa obtener todos los valores (uno por cada oferta)
# Mientras que la selección de un índice (\d+) significa que se le asigna ese valor (el mismo) a todas las ofertas
# 
Feature Values:             |* li\{"id":"demo"}\->3 div\{}\->1 p\{}\|
# 
# Luego sigue una línea en blanco, para mantener el orden
#

# 
# El título de la sección de información de cada convocatoria
#
Offer Structure:
# 
# Lista de los features que forman parte del id de la convocatoria
# 
ID_features:                |['descripción','título','requisitos']|
# 
# Sección de features a capturar, se define un número
# indeterminado de pares (Feature Names, Feature Values)
# 
Feature Names:              |* *\{"class":"job_box_title"}\|
Feature Values:             |* div\{"class":"job_box_information"}\|
Feature Names:              |['título']|
Feature Values:             |* h1\{"class":"job_description_title"}\->1 label\{}\|
